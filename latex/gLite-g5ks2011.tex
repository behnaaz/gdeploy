\subsection{gDeploy Script}
\begin{frame}
\frametitle{Introduction}
\framesubtitle{gDeploy Script}
	\begin{block}{Goal}
		\begin{itemize}
			\item Deploy a minimalist gLite site.
			\item composed by :
			\begin{itemize}
				\item a BDII element.
			  \item a Batch scheduler.
			  \item a Computing element.
			  \item Workers nodes.
			\end{itemize}
		\end{itemize}
	\end{block}
	\begin{block}{Script}
	  \begin{itemize}
			\item Written in ruby.
			\item avaliable on \url{http://sbadia.github.com/gdeploy/}
		\end{itemize}
	\end{block}
\end{frame}
