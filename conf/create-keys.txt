== Creation
* Si vous ne souhaitez pas créer de mot de passe, il faut alors enlever l’option -des3
openssl genrsa -des3 -out cle-ssl.key 1024
  pass-phrase: glite
openssl req -new -key cle-ssl.key > demande-ssl.csr
openssl genrsa -des3 -out cle-CA.key 1024
openssl req -new -key cle-CA.key -out demande-CA.csr
openssl x509 -in demande-CA.csr -out certif-CA.crt -req -signkey cle-CA.key -days 365
openssl x509 -req -in demande-ssl.csr -out certif-ssl.crt -CA certif-CA.crt -CAkey cle-CA.key -CAcreateserial -CAserial CA.srl
