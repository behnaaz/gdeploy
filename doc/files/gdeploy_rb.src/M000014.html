def queue_config()
  f = File.new("#{DIR}/queue.conf", "w")
  f.puts "#!/bin/sh\nqmgr << EOF\nset server scheduling = True\nset server acl_hosts = \#{batch}\n\#{$nodes.each do |node| \"set server acl_host += \#{node}\"end}\nset server managers = root@`hostname -f`\nset server operators = root@`hostname -f`\nset server default_queue = default\nset server scheduler_iteration = 600\nset server node_check_rate = 150\nset server tcp_timeout = 12\nset server poll_jobs = False\nset server log_level = 3\\nEOF\n"
  f.close
end
