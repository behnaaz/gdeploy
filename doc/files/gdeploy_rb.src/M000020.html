def queue_config(sname, wn)
  f = File.new("#{DIR}/conf/queue.conf", "w")
  f.puts "#!/bin/sh\nqmgr << EOF\nset server scheduling = True\nset server acl_hosts = *.\#{sname}.grid5000.fr\nset server managers = root@`hostname -f`\nset server operators = root@`hostname -f`\nset server default_queue = default\nset server scheduler_iteration = 600\nset server node_check_rate = 150\nset server tcp_timeout = 12\nset server poll_jobs = False\nset server log_level = 3\nset queue default Priority = 100\nset queue default max_queuable = 100\nset queue default max_running = 100\nset queue default resources_max.nodect = \#{wn.length - 1}\nset server query_other_jobs = True\nset server resources_default.cput = 01:00:00\nset server resources_default.neednodes = 1\nset server resources_default.nodect = \#{wn.length - 1}\nset server resources_default.nodes = 1\nset server default_node = 1#shared\nset queue default resources_default.nodes = nodes=1:ppn=1\\nEOF\n"
  f.close
end
