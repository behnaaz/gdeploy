def conf_batch(batch, cehost, sname)
  f = File.new("#{DIR}/conf/site-info-batch.def", "w")
  f.puts "## Site-info.def Batch\nBATCH_SERVER=\"\#{batch}\"\nCE_HOST=\"\#{cehost}\"\nCE_SMPSIZE=`grep -c processor /proc/cpuinfo`\nVOS=\"\#{sname}\"\nQUEUES=\"default\"\nDEFAULT_GROUP_ENABLE=\"\#{sname}\"\nUSERS_CONF=/opt/glite/yaim/etc/conf/users.conf\nGROUPS_CONF=/opt/glite/yaim/etc/conf/groups.conf\nWN_LIST=/opt/glite/yaim/etc/conf/wn-list.conf\nCONFIG_MAUI=\"yes\"\n"
  f.close
end
