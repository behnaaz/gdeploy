    def exchange_keys
      result = send_kexinit
      verify_server_key(result[:server_key])
      session_id = verify_signature(result)
      confirm_newkeys

      return { :session_id        => session_id, 
               :server_key        => result[:server_key],
               :shared_secret     => result[:shared_secret],
               :hashing_algorithm => digester }
    end
