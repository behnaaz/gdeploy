    def cleanup
      if @compressor
        @compressor.finish if !@compressor.finished?
        @compressor.close
      end

      if @decompressor
        # we call reset here so that we don't get warnings when we try to
        # close the decompressor
        @decompressor.reset
        @decompressor.close
      end

      @compressor = @decompressor = nil
    end
