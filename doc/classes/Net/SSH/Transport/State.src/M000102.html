    def initialize(socket, role)
      @socket = socket
      @role = role
      @sequence_number = @packets = @blocks = 0
      @cipher = CipherFactory.get("none")
      @block_size = 8
      @hmac = HMAC.get("none")

      @compressor = @decompressor = nil
      @next_iv = ""
    end
