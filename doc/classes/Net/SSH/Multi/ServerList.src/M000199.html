    def flatten
      result = @list.inject([]) do |aggregator, server|
        case server
        when Server then aggregator.push(server)
        when DynamicServer then aggregator.concat(server)
        else raise ArgumentError, "server list contains non-server: #{server.class}"
        end
      end

      result.uniq
    end
